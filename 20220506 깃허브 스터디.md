실수의 거듭 제곱값을 계산하는 프로그램

    int main(void)
    {
        int n,i;
        double r,result=1.0;

        printf("실수의 값을 입력하시오: ");
        scanf("%lf",&n);

        printf("거듭제곱횟수를 입력하시오: ");
        scanf("%d",&n);

        for(i=1;i<=n;i++)
        {
            result=result*r;
        }
    
        printf("결과값은 %lf",result);

        return 0;
    }
